def Analyze(url):
    """
    hash値が一部表記ゆれしちゃいそう
    """
    S = request.urlopen(url).read().decode()
    S = S.split("\n")
    ans = ""
    for s in S:
        if "textarea valuebox name=\"name\" rows" in s:
            s = s.split("
")[:-1]
            for i in s:
                ans += i.split(">")[-1] + "\n"
    ans1 = ans
    S = request.urlopen(url).read().decode()
    S = S.split("\n")
    ans = []
    going = 4
    for i, s in enumerate(S):
        if any([hash in s for hash in ["MD5", "SHA1", "SHA256", "SHA512"]]):
            ans.append(S[i+3].strip())
            going -= 1
        if not going:
            break
    return ans1, ans
